42 void* iface_new_N_Qubit_Decomposition( double* mtx_real, double* mtx_imag, int qbit_num, bool optimize_layer_num, int initial_guess_num ) {
43 
44  int matrix_size = Power_of_2(qbit_num);
45 
46  // combining real and imaginary parts of the matrix inti MKL complex matrix
47  int element_num = matrix_size*matrix_size;
48  Matrix Umtx = Matrix(matrix_size, matrix_size);
49 
50  //#pragma omp parallel for
51  for(int idx = 0; idx < element_num; idx++) {
52  Umtx[idx].real = mtx_real[idx];
53  Umtx[idx].imag = mtx_imag[idx];
54  }
55 
56  // setting the initial guess type
57  guess_type initial_guess;
58  if ( initial_guess_num==0 ) {
59  initial_guess = ZEROS;
60  }
61  else if ( initial_guess_num==1 ) {
62  initial_guess = RANDOM;
63  }
64  else if ( initial_guess_num==2 ) {
65  initial_guess = CLOSE_TO_ZERO;
66  }
67  else {
68  printf("Wrong initial guess\n");
69  exit(-1);
70  }
71 
72  // creating an instance of class N_Qubit_decomposition
73  N_Qubit_Decomposition<Sub_Matrix_Decomposition>* instance = new N_Qubit_Decomposition<Sub_Matrix_Decomposition>( Umtx, qbit_num, optimize_layer_num, initial_guess );
74 
75  return (void*)instance;
